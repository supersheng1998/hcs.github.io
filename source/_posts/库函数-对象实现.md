# C库函数
## memcpy
注意地址重叠情况
```C++
#include<iostream>
using namespace std;

void* memcpy2(void* src, void* dst, int size)
{
	if (dst == NULL || src == NULL)
		return NULL;
	char* t_src = nullptr;
	char* t_dst = nullptr;
	if ((src < dst) && ((char*)dst - (char*)src < size))//注意转换成char *型
	{
		t_src = (char*)src + size - 1;//减一，因为size=1时，表示复制自身，不是下一个
		t_dst = (char*)dst + size - 1;
		while (size--)
		{
			*t_dst-- = *t_src--;
		}
	}
	else
	{
		t_src = (char*)src;
		t_dst = (char*)dst;
		while (size--)
		{
			*t_dst++ = *t_src++;
		}
	}

	return dst;
}

int main()
{
	char a[] = "abcdefg";
	memcpy2(a, a+3, 4);
	cout << a << endl;

	system("pause");
	return 0;
}
```
