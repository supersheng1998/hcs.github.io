# shared_ptr和weak_ptr

##实现的一些要点
1. 一般是实现有参构造、拷贝构造、赋值构造、析构、重载*、->  
2. 引用计数要在堆区创建（防止其中一个实例析构就没了）

##代码
```C++
#include<iostream>
#include<string>
using namespace std;

//template<class T> class Weak_ptr;//为了用weak_ptr的lock()，来生成share_ptr2用，需要拷贝构造用

template<class T>
class share_ptr2
{
public:
	//friend class Weak_ptr<T>;
	//构造
	share_ptr2(T* _ptr) :ptr(_ptr), pcount(new int(1))
	{}
	//拷贝
	share_ptr2(const share_ptr2<T>& second):ptr(second.ptr),pcount(second.pcount)
	{
		++(*pcount);
	}
	//=赋值
	share_ptr2<T>& operator=(const share_ptr2<T>& second)
	{
		if (second.ptr != ptr)
		{
			if (--(*pcount) == 0)
			{
				delete ptr;//注意
				delete pcount;
			}
			this->ptr = second.ptr;
			this->pcount = second.pcount;
			++(*pcount);
		}
		return *this;
	}
	//->
	T *operator->()
	{
		return this->ptr;
	}
	
	//*
	T& operator*()
	{
		return *(this->ptr);
	}

	
	//返回引用计数
	int use_count() {
		return *pcount;
	}
	//析构
	~share_ptr2()
	{
		if (*pcount == 1)
		{
			delete pcount;
			delete ptr;
		}
		else
		{
			--(*pcount);
		}
	}

private:
	T* ptr;
	int* pcount;
};


template<class T>
class Weak_ptr
{
public:
	Weak_ptr(const share_ptr2<T>& second) :wptr(second.ptr)
	{}
	T& operator*()
	{
		return *wptr;
	}
	T* operator->()
	{
		return wptr;
	}
private:
	T* wptr;
};

```

## 测试用例
